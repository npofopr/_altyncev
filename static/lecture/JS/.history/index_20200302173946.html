addEventListener()

            linkAuth.addEventListener('click', function() {...})

отслеживаем клик на переменной linkAuth

            linkAuth.addEventListener('click', function() {
                console.log('клик');
            })
        
            $(linkAuth).on('click', function() {
                console.log('клик');
            })

Объект-событие. Метод preventDefault();

            linkAuth.addEventListener('click', function(e) {
                e.preventDefault();
                console.log('клик');
            })
        
            $(linkAuth).on('click', function(e) {
                e.preventDefault();
                console.log('клик');
            })
